## 概要
Flutterで作成されたオリジナルのスマホコントローラーアプリからの操作データをBLE経由で受信し、CANFDで流す基板です。
![コントローラーアプリ](/KisaragiBlog/Gallery/Wireless/WirelessController.png)

## 回路図
![回路図](/KisaragiBlog/Gallery/Wireless/Wireless_v6.png)
回路図はこちらになっています。小型化のためにCANFDデータ転送のためにメインマイコンにはSTM32G431KBT6を利用し、無線マイコンにはXIAO ESP32-S3を採用しています。基本的な仕組みはESPがスマホからのBLE操作データを受け取り、UARTでSTM32に送信、受信してそのままCANFDに送信、という流れです。CANトランシーバーのMCP2542FDはサンプル請求を使うと月1,2くらいの頻度で無料で3個もらえます。配線のことを考えてPA14側とPA0側にペリフェラルを集中させて、配線をしやすくしました。また、ESP32S3のUSB電源供給で逆流が生じないように、CANコネクタ→ショットキーダイオード→ESP32S3、とつなぐ設計にしています。通常はCAN用のXH4ピンから電源を持ってきて、無線実験をするときはUSBから供給という方向性になっています。

## ソフトウェア
Flutterアプリは、アンドロイド用にしか開発してません。また、スマホアプリ、Bluetoothを使うなど、セキュリティ的に保証できない部分が含まれてますので、一旦非公開とさせていただきます。現状プロジェクトデータのほとんどをPrivateリポジトリに上げています。どうしても欲しい方はDMなどで声かけてくだされば、セキュリティなどを保証しないという条件付きで提供する可能性があります。

マイコン周りのプログラムは以下の通りです。  
ESP32S3側はArduino IDEで書き込んでいます。

#include <BLEDevice.h>
#include <BLEServer.h>
#include <BLEUtils.h>
#include <BLE2902.h>
#include <Wire.h>

#define DEVICENAME "UART test"
#define INTERVAL 1000 //millisec

#define SERVICE_UUID           "任意のUUID"
#define CHARACTERISTIC_UUID_RX "任意のUUID"
#define CHARACTERISTIC_UUID_TX "任意のUUID"

BLEServer *pServer = NULL;
BLECharacteristic * pTxCharacteristic;
bool deviceConnected = false;
bool oldDeviceConnected = false;
bool newDataReceived = false;

// --- コントローラーのデータ格納用配列 ---
// 初期値 [0x1A(ヘッダ), btn1, btn2, joyLX, joyLY, joyRX, joyRY, rssi, 0x00(フッタ)]
uint8_t packetData[9] = {0x1A, 0, 0, 127, 127, 127, 127, 0, 0};

unsigned long lastPrintTime = 0;

class MyServerCallbacks: public BLEServerCallbacks {
  void onConnect(BLEServer* pServer) {
    deviceConnected = true;
    // Serial.println("** device connected");
  };

  void onDisconnect(BLEServer* pServer) {
    deviceConnected = false;
    // Serial.println("** device disconnected");
  }
};

class MyCallbacks: public BLECharacteristicCallbacks {
  void onWrite(BLECharacteristic *pCharacteristic) {
    uint8_t* rxValue = pCharacteristic->getData();
    size_t rxLength = pCharacteristic->getLength();

    // 9バイト以上受信した場合のみ処理
    if (rxLength >= 9) {
      bool isValidPacket = false;

      // パターン1: ヘッダが0x1Aのパケット
      if (rxValue[0] == 0x1A) {
        isValidPacket = true;
      }
      // パターン2: 全てFFで最後が00のパケット
      // (0xFF,0xFF,0xFF,0xFF,0xFF,0xFF,0xFF,0xFF,0x00)
      else if (rxValue[0] == 0xFF && rxValue[1] == 0xFF && rxValue[2] == 0xFF && 
               rxValue[3] == 0xFF && rxValue[4] == 0xFF && rxValue[5] == 0xFF && 
               rxValue[6] == 0xFF && rxValue[7] == 0xFF && rxValue[8] == 0x00) {
        isValidPacket = true;
      }

      // 有効なパケットであればバッファにコピー
      if (isValidPacket) {
        for (int i = 0; i < 9; i++) {
          packetData[i] = rxValue[i];
        }
        newDataReceived = true;
      }
    }
  }
};

void setup() {
  // PCモニタリング用のUSBシリアル
  Serial.begin(115200);
  
  // STM32通信用のハードウェアシリアル (XIAO ESP32S3のD6/D7ピンを使用)
  // begin(baud, config, RX_PIN, TX_PIN)
  // D7 = GPIO44(RX), D6 = GPIO43(TX)
  Serial1.begin(115200, SERIAL_8N1, 44, 43);
  
  BLEDevice::init(DEVICENAME);
  pServer = BLEDevice::createServer();
  pServer->setCallbacks(new MyServerCallbacks());
  BLEService *pService = pServer->createService(SERVICE_UUID);
  pTxCharacteristic = pService->createCharacteristic(CHARACTERISTIC_UUID_TX, BLECharacteristic::PROPERTY_NOTIFY);
  pTxCharacteristic->addDescriptor(new BLE2902());
  
  BLECharacteristic * pRxCharacteristic = pService->createCharacteristic(CHARACTERISTIC_UUID_RX, BLECharacteristic::PROPERTY_WRITE | BLECharacteristic::PROPERTY_WRITE_NR);
  pRxCharacteristic->setCallbacks(new MyCallbacks());

  pService->start();
  pServer->getAdvertising()->start();

  // Serial.println("start advertising");
}

void loop() {
  // 50ms間隔でSTM32へ送信（※現在は受信したタイミングでのみ送信）
  if (deviceConnected && newDataReceived) {
      newDataReceived = false;

      // --- パケット確認用のSerial出力 ---
      // Serial.print("Received Packet: ");
      // for (int i = 0; i < 9; i++) {
      //   // %02X で2桁の16進数（大文字）として出力し、ゼロ埋めする
      //   Serial.printf("%02X ", packetData[i]);
      // }
      // Serial.println(); // 改行
      // ----------------------------------

      // 9バイトの配列を一括で送信
      Serial1.write(packetData, 9);
      lastPrintTime = millis();
  }
  
  if (!deviceConnected && oldDeviceConnected) {
    delay(500); 
    pServer->startAdvertising(); 
    // Serial.println("start advertising");
    oldDeviceConnected = deviceConnected;
  }

  if (deviceConnected && !oldDeviceConnected) {
    oldDeviceConnected = deviceConnected;
  }
  
  delay(2);
}
